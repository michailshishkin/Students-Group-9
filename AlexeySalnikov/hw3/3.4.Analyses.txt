#/!bin/bash                                                               //Shebang указывает что скрипт должен выполняться оболочкой bash

options=(                                                                 //Объявление переменной и ее аргументов
	'1'
	"2"
	'3'
	'4'
	'5'
)
echo 'Please choose an option'                                            //Выводит строку 'Please choose an option' в терминал
select input in "${options[@]}"; do                                       //Ночало выбора аргумента 1 ... 5
	case "$input" in                                                  // Открытие оператора выбора аргументов
		"${options[0]}")                                           // Ввод аргумента 1
			ls                                                           с содержимым текущего каталога
			;;
		"${options[1]}")                                   // Ввод аргумента 2
			date $'+Time: %T\nDate: %D'                         с текущими временем и датой
			;;
		"${options[2]}")                                           // Ввод аргумента 3
			read -p 'What file do you wish to check? ' finput        //Спрашивает какой файл я желаю проверить и принимает значение с ввода
			if [[ -d $finput ]]; then                                 //Условный оператор. При выборе каталога
				format='%s is a directory.\n'                              выведет сообщение is a directory.
			elif [[ -f $finput ]]; then                               //Условный оператор. При выборе файла
				format='%s is a file.\n'                                    выведет сообщение is a file.
			else                                                 //Условный оператор. 
				format='%s does not exist.\n'                     При выборе объекта отсутствующего в текущем каталоге выведет сообщение does not exist.
			fi                                           //Условие закрыто
			printf "$format" "$finput"    
			;;
		"${options[3]}")                                         //Ввод аргумента 4
			read -p 'Please enter filename: ' binput         //Спрашивает имя файла  и принимает значение с ввода
			cp "$binput"{,.bak}                              //Копирует указанный файл с расширением .bak, если он существует.
			;;
		"${options[4]}")                                         //Ввод аргумента 5
			break                                            //Завершение выполнения сценария скрипта
			;;
		*)
			echo 'Not a valid option.'                        //Выводит сообщение Not a valid option если введеного аргумента нет в переменной
			;;
	esac                                         //Закрытие оператора выбора аргументов
done                                                    //Конец выбора аргументов
